function parseUrl() 
{
     // Get the query string
     href = location.href.split('#')[0]
     location.url = /\?.+/.exec(href)

     // create the url array
     location.url = location.url?(location.url[0].substring(1).split('&')):[];

     // loop over the tag pairs
     for (var i=0,len=location.url.length;i<len;i++)
	 {
	 	if(unescape(location.url[i]).indexOf('<') != -1 || unescape(location.url[i]).indexOf('>') != -1)
	 		continue;	// CSRF prevention
	 	if(unescape(location.url[i]).indexOf('(') != -1 || unescape(location.url[i]).indexOf(')') != -1)
	 		continue;	// CSRF prevention
       var nameValuePair = location.url[i].split(/=(.*)/);
       location.url[nameValuePair[0]]=unescape(nameValuePair[1]);
       // document.write(nameValuePair[0]+' '+nameValuePair[1]+'<br>'); 
    }
}

function convertUrl(url)
{
	parseUrl()
	if(location.url.wlUserLabel == undefined) return url
	if(url.indexOf('?') == -1)
		url = url + '?'
	else
		url = url + '&'
	return url + 'wlUserLabel=' + location.url.wlUserLabel + '&wlCookie=' + location.url.wlCookie + '&wlneCookie=' + db_cookie
}
